    immediate = {

            random_defender = {
                capital_scope = {
                    save_event_target_as = xl_crisis_clear_project_2_target
                }
            }
        }
        random_owned_fleet = {
            limit = {
                NOT = {
                    has_fleet_flag = xl_crisis_fleet_event_moved
                }
            }
            set_location = event_target:xl_crisis_clear_project_2_target
            set_fleet_flag = xl_crisis_fleet_event_moved
        }
    }

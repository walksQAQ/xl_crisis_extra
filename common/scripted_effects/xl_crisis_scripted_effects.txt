#舰队生成
xl_mid_crisis_fleet_create = {
    #初始三支舰队
    if = {
        limit = {
            $MODE$ = 1
        }
        create_fleet = {
            name = xl_mid_crisis_fleet_1
            settings = {
                can_upgrade = no
                can_change_composition = no
                is_boss = yes
                uses_naval_capacity = no
                spawn_debris = no
            }
            effect = {
                set_fleet_flag = is_xl_mid_crisis_event_create_fleet
            }
        }
        last_created_fleet = {
            create_ship = {

            }
        }
    }
    else_if = {
        limit = {
            $MODE$ = 2
        }
        create_fleet = {
            name = xl_mid_crisis_fleet_2
            settings = {
                can_upgrade = no
                can_change_composition = no
                is_boss = yes
                uses_naval_capacity = no
                spawn_debris = no
            }
            effect = {
                set_fleet_flag = is_xl_mid_crisis_event_create_fleet
            }
        }
        last_created_fleet = {
            create_ship = {

            }
        }
    }
    #雷场船
    else_if = {
        limit = {
            $MODE$ = SP
        }
        create_fleet = {
            name = xl_mid_crisis_fleet_sp
            settings = {
                can_upgrade = no
                can_change_composition = no
                is_boss = yes
                uses_naval_capacity = no
                spawn_debris = no
            }
            effect = {
                set_fleet_flag = is_xl_mid_crisis_event_create_fleet
                set_fleet_flag = xl_mid_crisis_sp_fleet
            }
        }
        last_created_fleet = {
            create_ship = {

            }
        }
    }
}

xl_crisis_sp_project_start = {
    capital_scope = {
        solar_system = {
            save_event_target_as = xl_crisis_clear_sp_project_target
        }
        enable_special_project = {
            name = $PROJECT$
            owner = this
            location = event_target:xl_crisis_clear_sp_project_target
        }
    }
}

set_xl_crisis_fleet_location = {
    while = {
        count = $TIME$
        country_event = {
            id = xl_crisis_clear.9
        }
    }
}
